#include <stdio.h>
void sortArray(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
int sockMerchant(int n, int ar[]) {
    int pairs = 0;
    sortArray(ar, n);
    for (int i = 0; i < n - 1; i++) {
        if (ar[i] == ar[i + 1]) { 
            pairs++;
            i++; 
        }
    }   
    return pairs;
}
int main() {
    int n;
    scanf("%d", &n);
    int ar[n];
    
    for (int i = 0; i < n; i++) {
        scanf("%d", &ar[i]);
    }
    
    printf("%d\n", sockMerchant(n, ar));
    return 0;
}
