#include <stdio.h>
#include <string.h>
int main() {
    int n, q;
    scanf("%d", &n);
    char s[1000][21]; 
    for (int i = 0; i < n; i++) {
        scanf("%s", s[i]);
    }
    scanf("%d", &q);
    for (int i = 0; i < q; i++) {
        char query[21];
        scanf("%s", query);
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (strcmp(s[j], query) == 0) {
                count++;
            }
        }
        printf("%d\n", count);
    }
    return 0;
}
