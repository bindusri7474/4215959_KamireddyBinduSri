#include <stdio.h>
#include <stdlib.h>

int diagonalDifference(int n, int **arr) {
    int primary_sum = 0, secondary_sum = 0;
    for (int i = 0; i < n; i++) {
        primary_sum += arr[i][i];             
        secondary_sum += arr[i][n - i - 1];   
    }
    return abs(primary_sum - secondary_sum);
}

int main() {
    int n;
    scanf("%d", &n);
    int **arr = (int **)malloc(n * sizeof(int *));
    for (int i = 0; i < n; i++) {
        arr[i] = (int *)malloc(n * sizeof(int));
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }
    int result = diagonalDifference(n, arr);
    printf("%d\n", result);
    for (int i = 0; i < n; i++) {
        free(arr[i]);
    }
    free(arr);
    return 0;
}
