#include <stdio.h>
void plusMinus(const int *arr, int n) {
    int counts[3] = {0}; 
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            counts[0]++;
        else if (arr[i] < 0)
            counts[1]++;
        else
            counts[2]++;
    }
    for (int i = 0; i < 3; i++) {
        printf("%.6f\n", (double)counts[i] / n);
    }
}
int main() {
    int n;
    scanf("%d", &n);
    int arr[n];  
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }   
    plusMinus(arr, n);
    return 0;
}
