#include <stdio.h>
#include <stdlib.h>

void buildFrequency(int arr[], int n, int freq[], int size) {
    for (int i = 0; i < n; i++) {
        freq[arr[i]]++;
    }
}
void printFrequency(int freq[], int size) {
    for (int i = 0; i < size; i++) {
        printf("%d ", freq[i]);
    }
    printf("\n");
}

int main() {
    int n;
    scanf("%d", &n);
    int *arr = (int *)malloc(n * sizeof(int));
    if (!arr) {
        printf("Memory allocation failed!\n");
        return 1;
    }
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    int range = 100; // Fixed range like your code
    int *freq = (int *)calloc(range, sizeof(int));
    if (!freq) {
        printf("Memory allocation failed!\n");
        free(arr);
        return 1;
    }
    buildFrequency(arr, n, freq, range);
    printFrequency(freq, range);
    free(arr);
    free(freq);
    return 0;
}
