#include <stdio.h>

unsigned int flippingBits(unsigned int n) {
    unsigned int result = 0;
    for (int i = 0; i < 32; i++) {
        if ((n & (1U << i)) == 0) {
            result |= (1U << i);
        }
    }
    return result;
}

int main() {
    int q;
    scanf("%d", &q);
    while (q--) {
        unsigned int num;
        scanf("%u", &num);
        printf("%u\n", flippingBits(num));
    }
    return 0;
}
