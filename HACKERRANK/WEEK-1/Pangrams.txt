#include <stdio.h>
#include <ctype.h>
#include <string.h>

int is_pangram(const char* s) {
    int found[26] = {0};
    const char *ptr = s;
    
    while (*ptr) {
        if (isalpha(*ptr)) {
            found[tolower(*ptr) - 'a'] = 1;
        }
        ptr++;
    }
    for (int i = 0; i < 26; i++) {
        if (!found[i])
            return 0;
    }
    return 1;
}

int main() {
    char str[1000];
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';

    if (is_pangram(str))
        printf("pangram\n");
    else
        printf("not pangram\n");

    return 0;
}
