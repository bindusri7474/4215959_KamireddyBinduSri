#include <stdio.h>
void sortArray(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
void miniMaxSum(int arr[5]) {
    sortArray(arr, 5);
    long long minSum = 0, maxSum = 0;
    for (int i = 0; i < 4; i++)
        minSum += arr[i];
    for (int i = 1; i < 5; i++)
        maxSum += arr[i];
    printf("%lld %lld\n", minSum, maxSum);
}
int main() {
    int arr[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }
    miniMaxSum(arr);
    return 0;
}
